#G --> C

#C --> G

#T --> A

#A --> U 

# Write your code here

myList=list(map(str,input()))

newList=[]

for n,i in enumerate(myList):

    if i=='G':

        newList.append('C')

    elif i=='C':

        newList.append('G')

    elif i=='T':

        newList.append('A')

    elif i=='A':

        newList.append('U')

    else:
        #Invalid Input as a string.
        del newList[:]
        newList.append("Invalid Input")
        break

#Printing the List Here

print(''.join(newList))

myList=list(map(str,input().split()))
for n,i in enumerate(myList):
    if i=='G':
        myList[n]='C'
        print(myList[n])
    elif i=='C':
        myList[n]='G'
        print(myList[n])
    elif i=='T':
        myList[n]='A'
        print(myList[n])
    elif i=='A':
        myList[n]='T'
        print(myList[n])
    else:
        print("Invalid Input As A String")
